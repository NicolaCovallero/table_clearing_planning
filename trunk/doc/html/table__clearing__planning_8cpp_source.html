480 void CTableClearingPlanning::refinePushingPose(Eigen::Vector3f &point, double safe_margin)
481 {
482  Eigen::Vector3f proj_eigen_point;
483 
484  double a = this->plane_coefficients.values[0];
485  double b = this->plane_coefficients.values[1];
486  double c = this->plane_coefficients.values[2];
487  double d = this->plane_coefficients.values[3];
488  double distance;
489  pcl::geometry::project(point,this->plane_origin,this->plane_normal,proj_eigen_point);
490  distance = (point - proj_eigen_point).norm();
491  bool safe_pose = true;
492  if((a*point[0] + b*point[1] + c*point[2] + d) > 0) // there is a contact with the table
493  {
494  distance = - distance;
495  safe_pose = false;
496  }
497  else if(distance < safe_margin)
498  {
499  safe_pose = false;
500  }
501  if(not safe_pose)
502  {
503  point -= this->plane_normal * (safe_margin - distance);
504  }
